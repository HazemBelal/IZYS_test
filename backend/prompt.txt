backend\scraper\calendarScraper.js
import fetch from 'node-fetch';
import * as cheerio from 'cheerio';

const BASE_URL = 'https://www.forexfactory.com';

/**
 * Function to scrape the economic calendar from ForexFactory based on a specific timeframe.
 */
async function scrapeForexFactory(timeframeUrl) {
    try {
        const response = await fetch(`${BASE_URL}${timeframeUrl}`);
        const body = await response.text();
        const $ = cheerio.load(body);

        const events = [];

        // Loop through each row of the calendar
        $('table.calendar__table tbody').each((_, tbody) => {
            let currentDay = null;

            $(tbody).find('tr').each((_, row) => {
                // Handle day-breakers to group events by day
                if ($(row).hasClass('calendar__row--day-breaker')) {
                    currentDay = $(row).text().trim(); // Capture the date for the day
                    return; // Skip to the next row
                }

                const time = $(row).find('td.calendar__time').text().trim();
                const currency = $(row).find('td.calendar__currency').text().trim();
                const event = $(row).find('td.calendar__event .calendar__event-title').text().trim();
                
                // Scrape the impact
                const impactElement = $(row).find('td.calendar__impact span');
                const impactClass = impactElement.attr('class') || ''; 
                const impactTitle = impactElement.attr('title') || 'Low Impact'; 
                
                let impact = 'low'; 
                if (impactClass.includes('icon--ff-impact-red')) {
                    impact = 'high';
                } else if (impactClass.includes('icon--ff-impact-ora')) {
                    impact = 'medium';
                } else if (impactClass.includes('icon--ff-impact-gra')) {
                    impact = 'non-economic';
                }

                const actual = $(row).find('td.calendar__actual span').text().trim();
                const actualClass = $(row).find('td.calendar__actual span').attr('class') || '';
                const actualTitle = $(row).find('td.calendar__actual span').attr('title') || '';

                const forecast = $(row).find('td.calendar__forecast span').text().trim();

                const previous = $(row).find('td.calendar__previous span').text().trim();
                const previousClass = $(row).find('td.calendar__previous span').attr('class') || '';
                const previousTitle = $(row).find('td.calendar__previous span').attr('title') || '';

                if (event) {
                    events.push({
                        date: currentDay || 'N/A', // Use 'N/A' as fallback if date is empty
                        time: time || 'N/A',
                        currency: currency || 'N/A',
                        event,
                        impact,  
                        actual: actual || '',
                        actualClass: actualClass || '',
                        actualTitle: actualTitle || '',
                        forecast: forecast || '',
                        previous: previous || '',
                        previousClass: previousClass || '',
                        previousTitle: previousTitle || '',
                    });
                }
            });
        });

        return events;
    } catch (error) {
        console.error('Error scraping ForexFactory:', error);
        return [];
    }
}

/**
 * Wrapper to handle timeframes for scraping like today, tomorrow, this week, etc.
 */
async function scrapeForexFactoryRange(timeframe) {
    let urlPart;
    switch (timeframe) {
        case 'today':
            urlPart = '/calendar?day=today';
            break;
        case 'tomorrow':
            urlPart = '/calendar?day=tomorrow';
            break;
        case 'thisWeek':
            urlPart = '/calendar?week=this';
            break;
        case 'nextWeek':
            urlPart = '/calendar?week=next';
            break;
        case 'thisMonth':
            urlPart = '/calendar?month=this';
            break;
        case 'nextMonth':
            urlPart = '/calendar?month=next';
            break;
        case 'yesterday':
            urlPart = '/calendar?day=yesterday';
            break;
        case 'lastWeek':
            urlPart = '/calendar?week=last';
            break;
        case 'lastMonth':
            urlPart = '/calendar?month=last';
            break;
        default:
            urlPart = '/calendar?day=today'; // Default to today if timeframe is not recognized
    }

    console.log(`Scraping for timeframe: ${timeframe} using URL: ${urlPart}`);
    return await scrapeForexFactory(urlPart);
}

export { scrapeForexFactoryRange };

backend\server\server.js
import express from 'express';
import cors from 'cors';
import { scrapeForexFactoryRange } from '../scraper/calendarScraper.js';

const app = express();
const PORT = process.env.PORT || 5000;

app.use(cors());

app.get('/api/calendar', async (req, res) => {
    const { timeframe } = req.query;

    if (!timeframe) {
        return res.status(400).json({ error: 'Please provide a valid timeframe.' });
    }

    try {
        const events = await scrapeForexFactoryRange(timeframe);
        res.json({ events });
    } catch (error) {
        console.error('Error fetching calendar data:', error);
        res.status(500).json({ error: 'Error fetching calendar data.' });
    }
});

app.listen(PORT, () => {
    console.log(`Server running on http://localhost:${PORT}`);
});
